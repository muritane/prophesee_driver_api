#include "utils/em_frame_generator.h"
namespace Prophesee {
EMFrameGenerator::EMFrameGenerator(long width, long height, bool process_all_frames) :
process_all_frames_(process_all_frames) {
ts_ = cv::Mat_<std::int32_t>(height, width, 0);
dts_ = cv::Mat_<std::int32_t>(height, width, std::numeric_limits<std::int32_t>::max());
frame_ = cv::Mat_<std::uint8_t>(height, width, std::uint8_t(0));
frame_swap_ = cv::Mat_<std::uint8_t>(height, width, std::uint8_t(0));
}
EMFrameGenerator::~EMFrameGenerator() {
stop();
}
void EMFrameGenerator::add_events(const Prophesee::EventEM *begin, const Prophesee::EventEM *end) {
if (thread_should_stop_ && !process_all_frames_) {
// if the generator is not yet started and we don't have to process all frames, we don't need to keep the events
return;
}
{
std::lock_guard<std::mutex> lock(processing_mutex_);
events_queue_front_.insert(events_queue_front_.end(), begin, end);
}
if ((end - 1)->t >= next_process_ts_) {
{
do {
next_process_ts_ += frame_period_;
} while ((end - 1)->t >= next_process_ts_);
std::unique_lock<std::mutex> lock(thread_cond_mutex_);
thread_should_process_ = true;
}
thread_cond_.notify_one();
}
}
void EMFrameGenerator::generate() {
Prophesee::timestamp next_frame_time = frame_period_;
while (true) {
{
std::unique_lock<std::mutex> lock(thread_cond_mutex_);
thread_cond_.wait(lock, [this] { return thread_should_stop_ || thread_should_process_; });
thread_should_process_ = false;
if (thread_should_stop_)
break;
}
{
std::unique_lock<std::mutex> lock(processing_mutex_);
std::swap(events_queue_front_, events_queue_back_);
}
if (events_queue_back_.empty()) {
continue;
}
if (process_all_frames_) {
auto last_ev_it = events_queue_back_.end();
auto it_begin = events_queue_back_.begin();
while (true) {
auto it_end = std::lower_bound(
it_begin, last_ev_it, next_frame_time,
[](const Prophesee::EventEM &ev, const Prophesee::timestamp &t) { return ev.t < t; });
update_frame(it_begin, it_end);
if (it_end != last_ev_it) { // if it_end == last_ev_it it means we are not finished processing the
// events for last frame
mapper_(dts_, frame_);
std::swap(frame_, frame_swap_);
// call the callback with updated frame
frame_cb_(next_frame_time, frame_swap_);
next_frame_time += frame_period_;
} else {
break;
}
it_begin = it_end;
}
} else {
Prophesee::timestamp frame_ts = events_queue_back_.back().t;
update_frame(events_queue_back_.begin(), events_queue_back_.end());
mapper_(dts_, frame_);
std::swap(frame_, frame_swap_);
// call the callback with updated frame
frame_cb_(frame_ts, frame_swap_);
}
events_queue_back_.clear();
}
}
void EMFrameGenerator::update_frame(std::vector<Prophesee::EventEM>::iterator it_begin,
std::vector<Prophesee::EventEM>::iterator it_end) {
for (auto it = it_begin; it != it_end; ++it) {
std::int32_t &t = ts_(it->y, it->x);
if (it->p == 0) {
t = it->t;
} else {
if (t < std::numeric_limits<std::int32_t>::max()) {
dts_(it->y, it->x) = it->t - t;
t = std::numeric_limits<std::int32_t>::max();
}
}
}
}
bool EMFrameGenerator::start(std::uint16_t fps,
const std::function<void(const Prophesee::timestamp &, const cv::Mat &)> &cb) {
if (thread_.joinable()) { // Already started
return false;
}
std::unique_lock<std::mutex> lock(thread_cond_mutex_);
frame_period_ = static_cast<Prophesee::timestamp>(1.e6 / fps + 0.5);
frame_cb_ = cb;
next_process_ts_ = frame_period_;
thread_should_process_ = false;
thread_should_stop_ = false;
thread_ = std::thread([this] { generate(); });
return true;
}
bool EMFrameGenerator::stop() {
if (!thread_.joinable()) {
return false;
}
{
std::unique_lock<std::mutex> lock(thread_cond_mutex_);
thread_should_stop_ = true;
}
thread_cond_.notify_one();
thread_.join();
return true;
}
