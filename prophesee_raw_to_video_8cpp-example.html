Example of using Prophesee Driver API for generating a video from a RAW file.

#include <prophesee_driver.h>
#include <utils/tone_mapper.h>
#include <iostream>
#include <boost/program_options.hpp>
#include <boost/filesystem.hpp>
#include <opencv2/core.hpp>
#include <opencv2/videoio.hpp>
#if CV_MAJOR_VERSION >= 4
#include <opencv2/highgui/highgui_c.h>
#endif
#include "cd_video_generator.h"
namespace po = boost::program_options;
void remove_file(const std::string &filepath) {
boost::filesystem::remove(boost::filesystem::path(filepath));
}
bool setup_graylevel_video_generator_callback(Prophesee::Camera &camera, cv::Mat &graylevel_frame,
Prophesee::Utils::GammaToneMapper &tone_mapper,
cv::VideoWriter &gray_level_video_writer, std::uint16_t fps) {
try {
camera.set_exposure_frame_callback(
fps, [&gray_level_video_writer, &tone_mapper, &graylevel_frame](Prophesee::timestamp ts, const cv::Mat &f) {
tone_mapper(f, graylevel_frame);
cv::cvtColor(graylevel_frame, graylevel_frame, CV_GRAY2BGR, 3);
gray_level_video_writer.write(graylevel_frame);
});
return true;
} catch (Prophesee::CameraException &e) {
if (e.code().value() & Prophesee::CameraErrorCode::UnsupportedFeature) {
std::cerr << "ERROR : EM not available" << std::endl;
}
std::cerr << e.what() << std::endl;
return false;
}
return false;
}
int main(int argc, char *argv[]) {
std::string input_file_path;
std::string cd_video_path("");
std::string graylevel_video_path("");
Prophesee::timestamp cd_accumulation_time;
std::uint16_t fps;
std::string fourcc;
po::options_description desc("\nTool to generate a video from a raw file.");
// clang-format off
desc.add_options()
("help,?", "produce help message")
("input,i", po::value<std::string>(&input_file_path)->required(), "Input RAW file.")
("output-cd", po::value<std::string>(&cd_video_path), "Output path of the CD events video.")
("output-gl", po::value<std::string>(&graylevel_video_path), "Output path of the gray levels (tone mapped frames computed from CD and EM) video.")
("cd-accumulation-time", po::value<Prophesee::timestamp>(&cd_accumulation_time)->default_value(10000), "Accumulation time (in us) for the CD video.")
("fps", po::value<std::uint16_t>(&fps)->default_value(30), "Frame rate of the video(s) to generate.")
("fourcc", po::value<std::string>(&fourcc)->default_value("MJPG"), "Fourcc 4-character code of codec used to compress the frames. List of codes can be "
"obtained at [Video Codecs by FOURCC](http://www.fourcc.org/codecs.php) page.")
;
// clang-format on
po::variables_map vm;
try {
po::store(po::command_line_parser(argc, argv).options(desc).run(), vm);
po::notify(vm);
} catch (po::error &e) {
std::cerr << "ERROR: " << e.what() << std::endl << std::endl;
std::cout << desc << std::endl;
return -1;
}
if (vm.count("help")) {
std::cout << desc << std::endl;
return 0;
}
// Check input arguments
if (cd_video_path.empty() && graylevel_video_path.empty()) {
std::cout << "No videos to generate" << std::endl;
return 0;
}
if (cd_video_path == graylevel_video_path) {
std::cerr << "ERROR: need to provide different output video paths for CD and EM" << std::endl;
return -1;
}
Prophesee::Camera camera;
try {
camera = Prophesee::Camera::from_file(input_file_path, false);
} catch (Prophesee::CameraException &e) {
std::cerr << e.what() << std::endl;
return -1;
}
// Get the geometry of the camera
auto &geometry = camera.geometry();
// Set up video for CD
CDVideoGenerator cd_video_generator(geometry.width(), geometry.height());
if (!cd_video_path.empty()) {
cd_video_generator.frame_generator().set_display_accumulation_time_us(cd_accumulation_time);
camera.cd().add_callback(
[&cd_video_generator](const Prophesee::EventCD *ev_begin, const Prophesee::EventCD *ev_end) {
cd_video_generator.add_events(ev_begin, ev_end);
});
if (!cd_video_generator.start(cd_video_path, fps, fourcc)) {
remove_file(cd_video_path);
return -1;
}
}
// Set up video for EM
Prophesee::Utils::GammaToneMapper tone_mapper;
cv::VideoWriter gray_level_video_writer;
cv::Mat graylevel_frame;
if (!graylevel_video_path.empty()) {
gray_level_video_writer.open(graylevel_video_path,
cv::VideoWriter::fourcc(fourcc[0], fourcc[1], fourcc[2], fourcc[3]), fps,
cv::Size(geometry.width(), geometry.height()));
bool gl_setup_success = true;
if (!gray_level_video_writer.isOpened()) {
std::cerr << "Error could not open file for writing at " << graylevel_video_path << std::endl;
gl_setup_success = false;
} else {
if (!(setup_graylevel_video_generator_callback(camera, graylevel_frame, tone_mapper,
gray_level_video_writer, fps))) {
gray_level_video_writer.release();
remove_file(graylevel_video_path);
gl_setup_success = false;
}
}
if (!gl_setup_success) {
if (!cd_video_path.empty()) {
cd_video_generator.stop();
remove_file(cd_video_path);
}
return -1;
}
}
// Start the camera streaming
const std::string message("Generating video(s) ");
std::cout << message;
std::cout.flush();
camera.start();
int dots = 0;
for (int it = 0; camera.is_running(); ++it) {
std::this_thread::sleep_for(std::chrono::milliseconds(10));
if (it == 50) {
it = 0;
dots = (dots + 1) % 4;
std::cout << "\r ";
std::cout << "\r" << message << std::string("...").substr(0, dots);
std::cout.flush();
}
}
std::cout << "\r";
// Stop the camera streaming
camera.stop();
// Stop the video generator
cd_video_generator.stop();
if (!cd_video_path.empty()) {
std::cout << "CD video has been saved in " << cd_video_path << std::endl;
}
if (!graylevel_video_path.empty()) {
std::cout << "Gray levels video has been saved in " << graylevel_video_path << std::endl;
}
return 0;
}
