Example of converting RAW file to DAT file

/*******************************************************************
* File : prophesee_raw_to_dat.cpp *
* Simple application to convert RAW file to DAT file(s). *
* *
* Copyright: (c) 2015-2019 Prophesee *
*******************************************************************/
#include <iostream>
#include <fstream>
#include <chrono>
#include <atomic>
#include <mutex>
#include <thread>
#include <type_traits>
#include <condition_variable>
#include <boost/program_options.hpp>
#include <prophesee_driver.h>
#include <utils/DAT_helper.h>
namespace po = boost::program_options;
template<typename EventType>
class DATFileWriter {
public:
static constexpr size_t EventSize = Prophesee::get_event_size<EventType>();
static constexpr size_t MinEventCountToProcess = 10000;
DATFileWriter() = default;
~DATFileWriter() {
thread_done_ = true;
thread_cond_.notify_one();
if (thread_.joinable())
thread_.join();
}
bool set_path(const std::string &path) {
fs_.open(path, std::ios::binary);
if (!fs_.is_open()) {
return false;
}
thread_done_ = false;
thread_ = std::thread([this]() {
while (!thread_done_) {
std::unique_lock<std::mutex> lock(buffer_safety_);
// wait for some events to be added in the buffer ..
thread_cond_.wait(lock, [this]() { return events_.size() > MinEventCountToProcess || thread_done_; });
// .. then converts and writes them in the file
if (!events_.empty())
convert_events();
if (!buffer_.empty())
write_buffer();
}
});
return true;
}
void write_header(int width, int height) {
Prophesee::write_DAT_header<EventType>(fs_, Prophesee::make_DAT_header_map_with_geometry(width, height));
}
void append_events(const EventType *first, const EventType *last) {
std::unique_lock<std::mutex> lock(buffer_safety_);
events_.insert(events_.end(), first, last);
// Make sure we have enough events to process before waking up the writing thread
if (events_.size() > MinEventCountToProcess)
thread_cond_.notify_one();
}
private:
void convert_events() {
size_t num_events = events_.size();
buffer_.resize(num_events * EventSize);
std::uint8_t *ptr = buffer_.data();
for (auto &ev : events_) {
ev.write_event(ptr, 0);
ptr += EventSize;
}
events_.clear();
}
void write_buffer() {
fs_.write((const char *)buffer_.data(), buffer_.size());
buffer_.clear();
}
std::ofstream fs_;
std::condition_variable thread_cond_;
std::thread thread_;
std::atomic<bool> thread_done_;
std::mutex buffer_safety_;
std::vector<EventType> events_;
std::vector<std::uint8_t> buffer_;
};
template<typename EventType, typename EventFacilityGetter>
bool setup_DAT_file_writer(DATFileWriter<EventType> &file_writer, EventFacilityGetter ev_facility_getter,
const std::string &file_path, int width, int height, size_t &num_events) {
bool facility_generated = false;
try {
auto &ev_facility = ev_facility_getter();
facility_generated = true;
if (!file_writer.set_path(file_path)) {
std::cerr << "Unable to write " << Prophesee::get_event_name<EventType>() << " DAT file." << std::endl;
return false;
}
file_writer.write_header(width, height);
ev_facility.add_callback([&](const EventType *first, const EventType *last) {
// add events in the buffer of events to be written and keep track of how many there are
file_writer.append_events(first, last);
num_events += std::distance(first, last);
});
} catch (Prophesee::CameraException &e) {
if (facility_generated) {
std::cerr << "Unable to set " << Prophesee::get_event_name<EventType>() << " callback." << std::endl;
} else {
std::cerr << "Unable to get the facility for event type: '" << Prophesee::get_event_name<EventType>()
<< "'." << std::endl;
}
return false;
}
return true;
}
int main(int argc, char *argv[]) {
std::string in = "";
bool do_cd = false;
bool do_em = false;
bool do_ext_trigger = false;
po::options_description desc("Allowed options");
// clang-format off
desc.add_options()("help,?", "produce help message")
("in,i", po::value<std::string>(&in)->default_value(""), "event raw file .raw")
("cd", po::bool_switch(&do_cd)->default_value(false), "generate _td.dat")
("em", po::bool_switch(&do_em)->default_value(false), "generate _aps.dat")
("trigger", po::bool_switch(&do_ext_trigger)->default_value(false), "generate _trigger.dat")
;
// clang-format on
po::positional_options_description p;
p.add("in", -1);
po::variables_map vm;
po::store(po::command_line_parser(argc, argv).options(desc).positional(p).run(), vm);
po::notify(vm);
if (vm.count("help")) {
std::cout << desc << "\n";
return 0;
}
if (!do_cd && !do_em && !do_ext_trigger) {
do_cd = true;
do_em = true;
}
// Open the device and flash firmware if requested
if (in == "") {
std::cerr << "Usage : " << argv[0] << std::endl;
std::cerr << desc << std::endl;
return 1;
}
Prophesee::Camera camera;
int width, height;
try {
camera = Prophesee::Camera::from_file(in, false);
width = camera.geometry().width();
height = camera.geometry().height();
} catch (Prophesee::CameraException &e) {
std::cerr << "Unable to open RAW file." << std::endl;
return 1;
}
size_t num_cd_events = 0, num_em_events = 0, num_ext_trigger_events = 0;
std::string out_base = in.substr(0, in.find_last_of(".raw") - 3);
{
DATFileWriter<Prophesee::EventCD> cd_file_writer;
DATFileWriter<Prophesee::EventEM> em_file_writer;
DATFileWriter<Prophesee::EventExtTrigger> ext_trigger_file_writer;
if (do_cd) {
do_cd = setup_DAT_file_writer(cd_file_writer, [&camera]() -> Prophesee::CD & { return camera.cd(); },
out_base + "_td.dat", width, height, num_cd_events);
}
if (do_em) {
do_em = setup_DAT_file_writer(em_file_writer, [&camera]() -> Prophesee::EM & { return camera.em(); },
out_base + "_aps.dat", width, height, num_em_events);
}
if (do_ext_trigger) {
do_ext_trigger = setup_DAT_file_writer(
ext_trigger_file_writer, [&camera]() -> Prophesee::ExtTrigger & { return camera.ext_trigger(); },
out_base + "_trigger.dat", width, height, num_ext_trigger_events);
}
std::cout << "Writing DAT files";
std::cout.flush();
camera.start();
int dots = 0;
for (int it = 0; camera.is_running(); ++it) {
std::this_thread::sleep_for(std::chrono::milliseconds(10));
if (it == 50) {
it = 0;
dots = (dots + 1) % 4;
std::cout << "\r ";
std::cout << "\rWriting DAT files" << std::string("...").substr(0, dots);
std::cout.flush();
}
}
std::cout << "\r";
camera.stop();
}
if (do_cd) {
std::cout << "Wrote " << num_cd_events << " " << Prophesee::get_event_name<Prophesee::EventCD>() << " events."
<< std::endl;
}
if (do_em) {
std::cout << "Wrote " << num_em_events << " " << Prophesee::get_event_name<Prophesee::EventEM>() << " events."
<< std::endl;
}
if (do_ext_trigger) {
std::cout << "Wrote " << num_ext_trigger_events << " "
<< Prophesee::get_event_name<Prophesee::EventExtTrigger>() << " events." << std::endl;
}
return 0;
}
