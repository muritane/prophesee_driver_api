#include "cd_video_generator.h"
#if CV_MAJOR_VERSION >= 4
#include <opencv2/highgui/highgui_c.h>
#endif
CDVideoGenerator::CDVideoGenerator(long width, long height) :
width_(width),
height_(height),
frame_generator_(width, height, true) {}
CDVideoGenerator::~CDVideoGenerator() {
stop();
}
Prophesee::CDFrameGenerator &CDVideoGenerator::frame_generator() {
return frame_generator_;
}
bool CDVideoGenerator::start(const std::string &output, std::uint16_t fps, const std::string &fourcc) {
frame_period_ = static_cast<Prophesee::timestamp>(1.e6 / fps + 0.5);
video_writer_.open(output, cv::VideoWriter::fourcc(fourcc[0], fourcc[1], fourcc[2], fourcc[3]), fps,
cv::Size(width_, height_));
if (!video_writer_.isOpened()) {
std::cerr << "Error could not open file for writing at " << output << std::endl;
return false;
}
return frame_generator_.start(
fps, [this](const Prophesee::timestamp &ts, const cv::Mat &frame) { add_frame(ts, frame); });
}
bool CDVideoGenerator::stop() {
bool stopped = frame_generator_.stop();
video_writer_.release();
return stopped;
}
void CDVideoGenerator::add_events(const Prophesee::EventCD *begin, const Prophesee::EventCD *end) {
frame_generator_.add_events(begin, end);
// Wait until corresponding frame is generated
Prophesee::timestamp last_ts = (end - 1)->t;
Prophesee::timestamp last_frame_ts_expected = last_ts - last_ts % frame_period_;
{
std::unique_lock<std::mutex> lock(thread_cond_mutex_);
thread_cond_.wait(lock, [this, last_frame_ts_expected] { return (last_frame_ts_ >= last_frame_ts_expected); });
}
}
void CDVideoGenerator::add_frame(const Prophesee::timestamp &ts, const cv::Mat &frame) {
cv::cvtColor(frame, cd_frame_, CV_GRAY2BGR, 3);
video_writer_.write(cd_frame_);
{
std::unique_lock<std::mutex> lock(thread_cond_mutex_);
last_frame_ts_ = ts;
}
thread_cond_.notify_one();
}
